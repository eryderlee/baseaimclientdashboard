3. Refactor `app/admin/preview/[clientId]/page.tsx`:
   - Strip out ALL the inline rendering (DashboardOverview, serialized milestones, etc.)
   - Replace with: verify ADMIN role, validate clientId exists, set both cookies (preview clientId and returnTo), then redirect.
   - The page should accept a `returnTo` search param to know where the admin came from:
     ```
     import { cookies } from 'next/headers'
     import { redirect } from 'next/navigation'
     import { verifySession } from '@/lib/dal'
     import { prisma } from '@/lib/prisma'

     export default async function AdminPreviewPage({
       params,
       searchParams,
     }: {
       params: Promise<{ clientId: string }>
       searchParams: Promise<{ returnTo?: string }>
     }) {
       const { userRole } = await verifySession()
       if (userRole !== 'ADMIN') redirect('/dashboard')

       const { clientId } = await params
       const { returnTo } = await searchParams
       const client = await prisma.client.findUnique({ where: { id: clientId }, select: { id: true } })
       if (!client) redirect('/admin')

       const cookieStore = await cookies()
       cookieStore.set('admin_preview_clientId', clientId, {
         path: '/',
         httpOnly: true,
         secure: process.env.NODE_ENV === 'production',
         sameSite: 'lax',
       })

       // Store the page admin came from so Exit Preview returns them there
       const safeReturnTo = returnTo && returnTo.startsWith('/') ? returnTo : '/admin'
       cookieStore.set('admin_preview_return_to', safeReturnTo, {
         path: '/',
         httpOnly: true,
         secure: process.env.NODE_ENV === 'production',
         sameSite: 'lax',
       })

       redirect('/dashboard')
     }
     ```
   - This is clean: visit `/admin/preview/[clientId]?returnTo=/admin/somepage` -> sets both cookies -> redirects to `/dashboard` -> dashboard reads cookie -> shows client data with banner. Exit Preview reads returnTo cookie and sends admin back.
